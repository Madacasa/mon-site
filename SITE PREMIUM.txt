  <script>
    const car = document.getElementById('car');
    const startDate = document.getElementById('startDate');
    const endDate = document.getElementById('endDate');
    const estimate = document.getElementById('estimate');
    const bookingForm = document.getElementById('bookingForm');

    function getDays() {
      const start = new Date(startDate.value);
      const end = new Date(endDate.value);
      const diff = end - start;
      const days = Math.ceil(diff / (1000 * 60 * 60 * 24));
      return days > 0 ? days : 0;
    }

    function updateEstimate() {
      const rate = Number(car.value || 0);
      const days = getDays();
      const total = rate * days;
      estimate.textContent = days > 0 && rate > 0
        ? `Montant estimé : ${total} MAD pour ${days} jour(s)`
        : 'Montant estimé : 0 MAD';
    }

    car.addEventListener('change', updateEstimate);
    startDate.addEventListener('change', updateEstimate);
    endDate.addEventListener('change', updateEstimate);

    bookingForm.addEventListener('submit', function(e){
      e.preventDefault();
      if(getDays() <= 0){
        alert('Veuillez choisir une date valide.');
        return;
      }
      alert('Votre demande a été envoyée.');
      bookingForm.reset();
      estimate.textContent = 'Montant estimé : 0 MAD';
    });
  </script>
